function sendTargetedPush(key, taskData, action) { var parts = key.split('_'); var yearK = parseInt(parts[0]); var weekK = parseInt(parts[1]); var user = parts[2] || ''; var dayIdx = parseInt(parts[3]); var days = ['Lundi','Mardi','Mercredi','Jeudi','Vendredi']; var dayName = days[dayIdx] || ''; var dateStr = fmtDate(getDateFromWeek(weekK, yearK, dayIdx)); var chan = taskData.chantier ? '['+taskData.chantier.toUpperCase()+']' : ''; var miss = taskData.text ? taskData.text.substring(0,50) : (taskData.isConge ? 'CONGE' : ''); var emoji = action === 'NOUVELLE' ? 'NEW' : (action === 'ANNULEE' ? 'ANNUL' : 'MODIF'); var title = 'RBF - Mission ' + action; var body = user.toUpperCase() + ' | ' + dayName + ' ' + dateStr + (chan ? ' | '+chan : '') + (miss ? ' | '+miss : ''); db.ref('push_subs').once('value').then(function(snap) { var data = snap.val() || {}; var subs = []; for (var u in data) { try { var s = JSON.parse(data[u].sub); if (s && s.endpoint) subs.push(s); } catch(e) {} } if (!subs.length) { console.log('Aucun abonne push'); return; } console.log('Envoi push a', subs.length, 'abonnes:', title); fetch(PUSH_WORKER_URL, { method : 'POST', headers: { 'Content-Type': 'application/json' }, body : JSON.stringify({ subscriptions: subs, title: title, message: body }) }) .then(function(r) { return r.json(); }) .then(function(d) { console.log('Push OK:', d.sent, 'envoyes,', d.errors, 'erreurs'); }) .catch(function(e) { console.warn('Push erreur:', e.message); }); }); }
